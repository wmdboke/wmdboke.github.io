## FTP 服务

### vsftpd 配置

```ini
# 允许匿名访问
anonymous_enable=NO

# 允许本地用户登录
local_enable=YES

# 允许写入
write_enable=YES

# 本地用户掩码
local_umask=022

# 允许匿名上传
anon_upload_enable=YES
anon_mkdir_write_enable=YES

# 显示目录消息
dirmessage_enable=YES

# 启用日志
xferlog_enable=YES

# 端口20
connect_from_port_20=YES

# 使用本地时间
use_localtime=YES

# PAM服务名
pam_service_name=vsftpd
userlist_enable=YES
tcp_wrappers=YES
```

#### 虚拟用户配置

```ini
listen=NO
listen_ipv6=YES
listen_port=21
idle_session_timeout=300
guest_enable=YES
guest_username=ftpuser
user_config_dir=/etc/vsftpd/vconf
data_connection_timeout=1
virtual_use_local_privs=YES
pasv_min_port=40000
pasv_max_port=40100
accept_timeout=5
connect_timeout=1
allow_writeable_chroot=YES
```

### lftp 配置

```ini
# 别名
alias dir ls
alias less more

# 提示符
set prompt "lftp \S\? \u\@\h:\w> "

# 同步模式
set sync-mode/ftp.idsoftware.com on

# 设置代理
set xfer:max-redirections 10

# FTP被动模式
# set ftp:passive-mode off
```

常用命令：
- `reconnect`: 重新连接
- `edit`: 编辑远程文件

## Keepalived 配置

```ini
! Configuration File for keepalived

global_defs {
   notification_email {
        clevercode@qq.com
        clevercode1@qq.com
        clevercode2@qq.com
  }
  notification_email_from root
  smtp_server 127.0.0.1
  smtp_connect_timeout 30
  router_id master-node
}

vrrp_script chk_nginx {
   script "/etc/keepalived/nginx_check.sh"
   interval 2
   weight -20
}

vrrp_instance VI_1 {
   state MASTER
   interface eth0
   virtual_router_id 140
   priority 100
   nopreempt
   advert_int 1

   authentication {
       auth_type PASS
       auth_pass nginx
   }

   track_script {
       chk_nginx
   }

   virtual_ipaddress {
       172.16.159.140
   }
   
   notify_master "/etc/keepalived/notify.sh master"
   notify_backup "/etc/keepalived/notify.sh backup"
   notify_fault  "/etc/keepalived/notify.sh fault"
}
```
